import { kv } from '@vercel/kv';

export default async function handler(req, res) {
  const { code } = req.query;
  const data = await kv.get(code);

  if (!data) {
    return res.status(400).json({ success: false, message: "无效或过期" });
  }
  if (data.usedCount >= 3) {
    return res.status(400).json({ success: false, message: "次数耗尽" });
  }

  await kv.set(code, { usedCount: data.usedCount + 1 }, { keepTtl: true });
  res.status(200).json({ success: true, message: "验证通过" });
}
